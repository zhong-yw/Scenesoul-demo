# 测试方案设计：白界 · Scenesoul

> **目标：** 为 v0.1 Demo 建立完整的测试体系，覆盖所有模块的核心逻辑
> **原则：** 纯函数优先测试 → 需 mock 的函数黑盒测试 → 集成测试验证交互
> **工具：** pytest + unittest.mock（Python 内置）
> **预估文件：** 约 15 个测试文件，分 6 个测试模块

---

## 四、核心测试用例代码示例

### 4.1 Token 估算

```python
def test_estimate_tokens_chinese():
    """纯中文：每个中文字符约 1.5 token"""
    # "你好世界" = 4 个中文字符
    # 预期: int(4 * 1.5 + 0 * 0.4) + 2 = 6 + 2 = 8
    assert _estimate_tokens("你好世界") == 8

def test_estimate_tokens_mixed():
    """中英混合"""
    text = "hello 世界"
    # chinese=2 → 2*1.5=3, other=5 → 5*0.4=2, int(3+2)+2 = 7
    assert _estimate_tokens("hello 世界") == 7

def test_estimate_messages_tokens():
    msg = [{"role": "user", "content": "你好"}, {"role": "assistant", "content": "世界"}]
    # user: content=2→(int(2*1.5)+2)=5, role=4→int(4*0.4)+2=3
    # assistant: content=2→5, role=9→int(9*0.4)+2=5
    # total = 5+3+5+5 = 18
    assert _estimate_messages_tokens(msg) == 6 + 6  # simplified, exact values TBD
```

### 4.2 界说标记解析

```python
def test_handle_new_scene_basic():
    agent = NarratorAgent(mock_llm)  # mock_llm not needed for this internal method
```

### 4.3 驱动力阈值

```python
def test_build_drive_prompts_hunger_thresholds():
    memory = BrainMemory()
    builder = BrainContextBuilder(memory)
    
    # hunger=20 → 低于 25，不触发
    memory.l1["drives"]["hunger"] = 20
    assert builder._build_drive_prompts() == ["你有点累了"]  # only fatigue from init
    
    # hunger=30 → 触发"你有点饿了"
    memory.l1["drives"]["hunger"] = 30
    hints = builder._build_drive_prompts()
    assert any("饿了" in h for h in hints)
    
    # hunger=60 → 触发更强烈的提示
    memory.l1["drives"]["hunger"] = 60
    hints = builder._build_drive_prompts()
    assert any("咕咕叫" in h for h in hints)
```

### 4.4 BrainAgent 错误处理

```python
def test_internal_think_llm_error(brain_agent):
    """LLM 异常时返回 fallback 并追加到上下文"""
    brain_agent.llm.chat.side_effect = Exception("API error")
    
    result = brain_agent.internal_think()
    
    assert result == "……（静静沉思着）"
    # 验证 assistant 消息被追加（保持消息对齐）
    assert len(brain_agent.ctx.messages) >= 2  # system + fallback assistant
    assert brain_agent.ctx.messages[-1]["role"] == "assistant"
```
